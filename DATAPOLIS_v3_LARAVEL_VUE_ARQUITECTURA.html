Controller Template

// app/Http/Controllers/Api/V1/PropTech/ExpedienteController.php

class ExpedienteController extends Controller
{
    public function __construct(private ExpedienteService $service) {}
    
    public function index(Request $request): PredioCollection
    {
        return new PredioCollection($this->service->list(
            $request->only(['comuna_id', 'destino', 'search']),
            $request->integer('per_page', 15)
        ));
    }
    
    public function store(StorePredioRequest $request): JsonResponse
    {
        $predio = $this->service->create($request->validated());
        return response()->json(['data' => new PredioResource($predio)], 201);
    }
    
    public function show(Predio $predio): PredioResource
    {
        $this->authorize('view', $predio);
        return new PredioResource($predio->load(['propietarios', 'valorizaciones']));
    }
    
    public function syncSII(Predio $predio): JsonResponse
    {
        $resultado = $this->service->syncWithSII($predio);
        return response()->json(['message' => 'Sincronizado', 'data' => $resultado]);
    }
    
    public function exportPdf(Predio $predio)
    {
        return $this->service->exportToPdf($predio);
    }
}

Model con Traits

// app/Models/PropTech/Predio.php

class Predio extends Model
{
    use HasFactory, HasUuids, SoftDeletes;
    use HasTenant, HasVersions, HasDocuments, HasAuditLog;
    use HasSpatial;
    
    protected $fillable = [
        'tenant_id', 'rol_sii', 'direccion', 'comuna_id',
        'superficie_terreno', 'superficie_construida', 'destino',
        'avaluo_fiscal', 'coordenadas', 'datos_sii', 'ultimo_sync_sii'
    ];
    
    protected $casts = [
        'coordenadas' => Point::class,
        'avaluo_fiscal' => 'decimal:2',
        'datos_sii' => 'array',
        'ultimo_sync_sii' => 'datetime'
    ];
    
    public function propietarios() { return $this->hasMany(Propietario::class); }
    public function valorizaciones() { return $this->hasMany(Valorizacion::class); }
    public function plusvalias() { return $this->hasMany(Plusvalia::class); }
    
    public function scopeNearby($query, Point $punto, int $metros = 1000)
    {
        return $query->whereRaw(
            'ST_DWithin(coordenadas::geography, ST_SetSRID(ST_MakePoint(?, ?), 4326)::geography, ?)',
            [$punto->longitude, $punto->latitude, $metros]
        );
    }
}

HasTenant Trait

// app/Models/Traits/HasTenant.php

trait HasTenant
{
    protected static function bootHasTenant(): void
    {
        // Auto-asignar tenant_id
        static::creating(function ($model) {
            if (!$model->tenant_id && auth()->check()) {
                $model->tenant_id = auth()->user()->tenant_id;
            }
        });
        
        // Global scope para filtrar por tenant
        static::addGlobalScope('tenant', function (Builder $builder) {
            if (auth()->check() && !app()->runningInConsole()) {
                $builder->where($builder->getModel()->getTable() . '.tenant_id', auth()->user()->tenant_id);
            }
        });
    }
    
    public function tenant() { return $this->belongsTo(Tenant::class); }
}

Pinia Store

// resources/js/stores/proptech/predio.js

export const usePredioStore = defineStore('predio', () => {
  const predios = ref([])
  const loading = ref(false)
  const pagination = ref({ current_page: 1, per_page: 15, total: 0 })
  
  async function fetchPredios() {
    loading.value = true
    try {
      const response = await api.get('/api/v1/predios', { params: { page: pagination.value.current_page } })
      predios.value = response.data.data
      pagination.value = response.data.meta
    } finally {
      loading.value = false
    }
  }
  
  async function createPredio(data) {
    const response = await api.post('/api/v1/predios', data)
    predios.value.unshift(response.data.data)
    return response.data.data
  }
  
  return { predios, loading, pagination, fetchPredios, createPredio }
})

8. 🛠️ Guía de Instalación

A. Docker (Local/Server)

cp .env.example .env

# 2. Crear BD MySQL en cPanel
# 3. Configurar .env:
APP_URL=https://tudominio.cl
DB_CONNECTION=mysql
DB_HOST=localhost
DB_DATABASE=datapolis_db
DB_USERNAME=datapolis_user
CACHE_DRIVER=file
QUEUE_CONNECTION=database

# 5. Cron job para queue:
